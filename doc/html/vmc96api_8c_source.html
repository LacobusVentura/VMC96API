480 static int vmc96_prepare_k1_message( VMC96_t * vmc96 )
481 {
482  vmc96->message.k1_length = vmc96->message.data_length + VMC96_K1_MESSAGE_MIN_LEN;
483 
484  /* K1 Message STX Header Field */
485  vmc96->message.k1[0] = VMC96_K1_MESSAGE_STX;
486 
487  /* K1 Message: Controller Address/ID Field */
488  vmc96->message.k1[1] = vmc96->message.id_controller;
489 
490  /* K1 Message: Total Length Field */
491  vmc96->message.k1[2] = vmc96->message.k1_length;
492 
493  /* K1 Message: Command Code Field */
494  vmc96->message.k1[3] = vmc96->message.command;
495 
496  /* K1 Message: Data Field */
497  if( vmc96->message.data_length > 0 )
498  memcpy( &vmc96->message.k1[4], vmc96->message.data, vmc96->message.data_length );
499 
500  /* K1 Message: Checksum Field */
501  vmc96->message.k1[ vmc96->message.k1_length - 1 ] = vmc96_calculate_checksum( vmc96->message.k1, vmc96->message.k1_length - 1 );
502 
503  return VMC96_SUCCESS;
504 }

559 static int vmc96_parse_k1_response( VMC96_t * vmc96 )
560 {
561  switch( vmc96_k1_parse_response_type( vmc96 ) )
562  {
563  case VMC96_K1_RESPONSE_TYPE_ACK:
564  {
565  /* K1 Response: Validate Positive ACK Message Len */
566  if( vmc96->response.k1_length != VMC96_K1_MESSAGE_MIN_LEN )
567  return VMC96_ERROR_K1_RESPONSE_INVALID_LENGTH;
568 
569  /* K1 Response: Parse Source Controller ID/Address Field */
570  vmc96->response.id_controller = vmc96->response.k1[1];
571 
572  /* K1 Response: Data Field Empty */
573  memset( vmc96->response.data, 0, VMC96_K1_MESSAGE_DATA_MAX_LEN );
574  vmc96->response.data_length = 0;
575 
576  /* K1 Response: Validating STX Header Field */
577  if( vmc96->response.k1[0] != VMC96_K1_MESSAGE_STX )
578  return VMC96_ERROR_K1_RESPONSE_MALFORMED;
579 
580  /* K1 Response: Validate Source Controller ID/Address Field */
581  if( vmc96->response.k1[1] != vmc96->message.id_controller )
582  return VMC96_ERROR_K1_RESPONSE_INVALID_SOURCE;
583 
584  /* K1 Response: Validate Positive ACK Message Len */
585  if( vmc96->response.k1[2] != VMC96_K1_MESSAGE_MIN_LEN )
586  return VMC96_ERROR_K1_RESPONSE_INVALID_LENGTH;
587 
588  /* K1 Response: Validate Positive ACK Field */
589  if( vmc96->response.k1[3] != VMC96_K1_RESPONSE_POSITIVE_ACK )
590  return VMC96_ERROR_K1_RESPONSE_NEGATIVE_ACK;
591 
592  /* K1 Response: Validate Checksum */
593  if( vmc96->response.k1[4] != vmc96_calculate_checksum( vmc96->response.k1, vmc96->response.k1_length - 1 ) )
594  return VMC96_ERROR_K1_RESPONSE_INVALID_CHECKSUM;
595 
596  return VMC96_SUCCESS;
597  }
598 
599  case VMC96_K1_RESPONSE_TYPE_DATA:
600  {
601  /* K1 Response: Validating Message Length */
602  if( vmc96->response.k1_length < VMC96_K1_MESSAGE_MIN_LEN )
603  return VMC96_ERROR_K1_RESPONSE_INVALID_LENGTH;
604
