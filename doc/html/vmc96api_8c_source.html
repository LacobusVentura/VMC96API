370 static int vmc96_parse_raw_response( VMC96_t * vmc96 )
371 {
372  switch( vmc96->message.id_controller )
373  {
374  case VMC96_CONTROLLER_GLOBAL_BROADCAST:
375  {
376  if( vmc96->response.raw_length <= 0 )
377  return VMC96_ERROR_RESPONSE_INVALID_LENGTH;
378 
379  vmc96->response.data_length = vmc96->response.raw_length;
380  memcpy( vmc96->response.data, &vmc96->response.raw, vmc96->response.raw_length );
381 
382  break;
383  }
384 
385  case VMC96_CONTROLLER_RELAY_1 :
386  case VMC96_CONTROLLER_RELAY_2 :
387  case VMC96_CONTROLLER_MOTOR_ARRAY:
388  {
389  if( vmc96->response.raw_length < 5 )
390  return VMC96_ERROR_RESPONSE_INVALID_LENGTH;
391 
392  vmc96->response.id_controller = vmc96->response.raw[1];
393  vmc96->response.data_length = vmc96->response.raw[2] - 4;
394  memcpy( vmc96->response.data, &vmc96->response.raw[3], vmc96->response.data_length );
395 
396  if( vmc96->response.raw[0] != VMC96_MESSAGE_STX )
397  return VMC96_ERROR_RESPONSE_MALFORMED;
398 
399  if( vmc96->response.raw[1] != vmc96->message.id_controller )
400  return VMC96_ERROR_RESPONSE_INVALID_SOURCE;
401 
402  if( vmc96->response.raw[2] != vmc96->response.raw_length )
403  return VMC96_ERROR_RESPONSE_INVALID_LENGTH;
404 
405  /*
406  if( vmc96->response.raw[3] != VMC96_POSITIVE_ACK_CODE )
407  return VMC96_ERROR_RESPONSE_NEGATIVE_ACK;
408  */
409 
410  if( vmc96->response.raw[ vmc96->response.raw_length - 1 ] != vmc96_calculate_checksum( vmc96->response.raw, vmc96->response.raw_length - 1 ) )
411  return VMC96_ERROR_RESPONSE_INVALID_CHECKSUM;
412 
413  break;
414  }
415 
416  default:
417  {
418  break;
419  }
420  }
421 
422  return VMC96_SUCCESS;
423 }

427 {

446 
447  vmc96->response.raw_length = ret;
