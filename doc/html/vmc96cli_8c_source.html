497 static int vmc96cli_proccess_arguments( int argc, char ** argv, vmc96cli_arguments_t * args )
498 {
499  int ret = 0;
500  int index = 0;
501 
502  static struct option options[] =
503  {
504  {"controller", required_argument, 0, 'a' },
505  {"cntlr", required_argument, 0, 'a' },
506  {"command", required_argument, 0, 'b' },
507  {"cmd", required_argument, 0, 'b' },
508  {"state", required_argument, 0, 'c' },
509  {"row", required_argument, 0, 'd' },
510  {"col", required_argument, 0, 'e' },
511  {"column", required_argument, 0, 'e' },
512  {"col1", required_argument, 0, 'f' },
513  {"column1", required_argument, 0, 'f' },
514  {"col2", required_argument, 0, 'g' },
515  {"column2", required_argument, 0, 'g' },
516  {"help", no_argument, 0, 'h' },
517  {0, 0, 0, 0 }
518  };
519 
520  args->controller = VMC96CLI_ARGUMENT_NOT_INITIALIZED;
521  args->command = VMC96CLI_ARGUMENT_NOT_INITIALIZED;
522  args->state = VMC96CLI_ARGUMENT_NOT_INITIALIZED;
523  args->row = VMC96CLI_ARGUMENT_NOT_INITIALIZED;
524  args->col = VMC96CLI_ARGUMENT_NOT_INITIALIZED;
525  args->col1 = VMC96CLI_ARGUMENT_NOT_INITIALIZED;
526  args->col2 = VMC96CLI_ARGUMENT_NOT_INITIALIZED;
527 
528  while(true)
529  {
530  ret = getopt_long( argc, argv, "a:b:c:d:e:f:g:h", options, &index );
531 
532  if( ret == -1 )
533  return VMC96CLI_SUCCESS;
534 
535  switch( ret )
536  {
537  case 'a' : args->controller = vmc96cli_get_cntrl_code( optarg ); break;
538  case 'b' : args->command = vmc96cli_get_cmd_code( optarg ); break;
539  case 'c' : args->state = atoi( optarg ); break;
540  case 'd' : args->row = atoi( optarg ); break;
541  case 'e' : args->col = atoi( optarg ); break;
542  case 'f' : args->col1 = atoi( optarg ); break;
543  case 'g' : args->col2 = atoi( optarg ); break;
544 
545  case 'h' :
546  vmc96cli_show_usage( argc, argv );
547  return -1;
548 
549  default :
550  return -1;
551  }
552  }
